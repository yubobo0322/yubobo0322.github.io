\begin{itemize}
	\item[1. ] pdf跳转到tex对应位置：\\ctrl和鼠标左键（mac：command和鼠标左键）；

    \right]=
    \left[ 
    \begin{array}{cc}
    cos \theta & sin \theta \\
    - sin \theta & cos \theta 
    \end{array}
    \right]
    \cdot
    \left[ 
    \begin{array}{c}
        x\\
        y\\
    \end{array}
    \right]
\end{equation}
